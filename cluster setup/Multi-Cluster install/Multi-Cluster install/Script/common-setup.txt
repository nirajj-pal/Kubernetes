#!/bin/bash

set -e

echo "[INFO] Disabling swap and updating system..."
swapoff -a
sed -i '/ swap / s/^/#/' /etc/fstab

apt update && apt upgrade -y

echo "[INFO] Installing containerd and setting up Kubernetes repo..."
apt install -y containerd apt-transport-https curl

mkdir -p /etc/containerd
containerd config default | tee /etc/containerd/config.toml
systemctl restart containerd
systemctl enable containerd

curl -s https://packages.cloud.google.com/apt/doc/apt-key.gpg | apt-key add -
echo "deb https://apt.kubernetes.io/ kubernetes-xenial main" \
    > /etc/apt/sources.list.d/kubernetes.list

apt update
apt install -y kubelet kubeadm kubectl
apt-mark hold kubelet kubeadm kubectl

echo "[INFO] Enabling kernel modules..."
modprobe overlay
modprobe br_netfilter

cat <<EOF | tee /etc/modules-load.d/k8s.conf
br_netfilter
overlay
EOF

cat <<EOF | tee /etc/sysctl.d/k8s.conf
net.bridge.bridge-nf-call-ip6tables = 1
net.bridge.bridge-nf-call-iptables  = 1
net.ipv4.ip_forward = 1
EOF

sysctl --system

echo "nameserver 10.48.216.132" > /etc/resolv.conf
